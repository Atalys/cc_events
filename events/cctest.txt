							}
						}
					}

					NOT = { has_planet_flag = cc_su_perf_planet_sowing_unrest }							prevprev = {
						any_owned_planet = {

								limit = { prevprevprev = { has_technology = tech_ccrebel_unrest_2 } }
								distance = {
									source = prevprev
									max_distance = 150
								}
								else = {
