country_event = {
	id = cctest.3
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		
		random_country = {
			limit = {
				is_country_type = default
				any_owned_planet = {
					is_neighbor_of = prevprev
				}
			}
			save_event_target_as = enemy
			root = {
				declare_war = {
					who = prev
				}
				add_war_demand = {
					type = "humiliate"
					parameter:country = prev
					warscore_cost = 10
					target = this
					enemy = prev
				}
				add_war_demand = {
					type = "dpe_cede_borders"
					parameter:country = prev
					warscore_cost = 90
					target = this
					enemy = prev
				}
				country_event = { id = dpe_diplo.100 scopes = { from = root } }
			}
		}
	}
}
