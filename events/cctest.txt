					}
				}
			}
			
			root = {
				if = {
					limit = {
						count_pops = {
							limit = {

						which = cctest_count_pops_1

					}
				}
				if = {
					limit = {
						count_pops = {
							limit = {
								is_same_species = prev
								log = [this.GetName]
								log = [this.planet.GetName]
								log = [prev.GetName]
								log = [planet.GetName]
								log = [owner.GetName]
							}
							count > 3
						}
					}
					set_variable = {
						which = cctest_count_pops_2
						value = 10
					}
				}
				if = {
					limit = {
						count_pops = {
							limit = {
								is_same_species = this

						}
